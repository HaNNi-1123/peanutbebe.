<script>

    // 【★★★ 請把下面這串換成妳剛才複製的通道鑰匙網址 ★★★】

    // 【★★★ 注意：網址前後的單引號 ' 不要刪掉喔 ★★★】

    const GOOGLE_SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbwtocFv4v1vZYCEu8m1FKiYtsgmf3-3kzl1A9E1U8lzmt_jrh5kItgtpilDm8HnbOKY7g/exec';


    const dateDropdown = document.getElementById('dateSelect');

    const today = new Date();

    

    for(let i=0; i<7; i++) {

        let d = new Date(today);

        d.setDate(today.getDate() - i);

        let yyyy = d.getFullYear();

        let mm = String(d.getMonth() + 1).padStart(2, '0');

        let dd = String(d.getDate()).padStart(2, '0');

        let dateString = `${yyyy}/${mm}/${dd}`;

        

        let opt = document.createElement('option');

        opt.value = dateString;

        if(i === 0) opt.innerText = `${dateString} (今天)`;

        else if(i === 1) opt.innerText = `${dateString} (昨天)`;

        else opt.innerText = dateString;

        dateDropdown.appendChild(opt);

    }


    // 核心處理函數：先送資料到 Google，再產出文案

    function processData() {

        const agent = document.getElementById('agentSelect').value;

        const remitDate = document.getElementById('dateSelect').value;

        const amount = document.getElementById('amountInput').value;

        const note = document.getElementById('noteInput').value;


        if(!agent) return alert('親愛的，選個代理編號再按送出。');

        if(!amount) return alert('儲值金額總不能是空的吧。');


        // 開啟 Loading 遮罩

        document.getElementById('loadingOverlay').style.display = 'flex';


        // 整理要送出的包裹

        const payload = {

            date: remitDate,

            agent: agent,

            amount: amount,

            note: note ? note : '無備註'

        };


        // 呼叫 Google 專屬快遞員

        fetch(GOOGLE_SCRIPT_URL, {

            method: 'POST',

            mode: 'no-cors', // 避免跨網域錯誤的強勢解法

            headers: {

                'Content-Type': 'application/json'

            },

            body: JSON.stringify(payload)

        })

        .then(() => {

            // 不管 Google 回什麼，只要送出去了，我們就立刻產出文案

            document.getElementById('loadingOverlay').style.display = 'none';

            generateTextContent(agent, remitDate, amount, payload.note);

        })

        .catch(error => {

            document.getElementById('loadingOverlay').style.display = 'none';

            alert('糟糕，同步到試算表時卡住了。但文案還是會幫妳產出喔。');

            generateTextContent(agent, remitDate, amount, payload.note);

        });

    }


function generateTextContent(agent, remitDate, amount, note) { let formattedAmount = Number(amount).toLocaleString(); // 這裡就是產出給代理複製回傳的文案 let finalString = `🥜 Peanuts.bebe 匯款回報單 🥜\n\n闆娘，我已完成儲值匯款囉！\n----------------------\n● 代理編號：${agent}\n● 匯款日期：${remitDate}\n● 匯款金額：NT$ ${formattedAmount}\n● 帳號末五碼：${note}\n----------------------\n請記得將此訊息【截圖】或【複製】回傳給闆娘核對喔！✨`; document.getElementById('outputText').innerText = finalString; document.getElementById('resultArea').style.display = 'block'; // 這裡是按鈕按下去後的彈窗提醒 alert('文案已產出！請記得「點擊下方按鈕」複製文案，並回傳給闆娘喔！'); }

    }


    function copyToClipboard() {

        const textToCopy = document.getElementById('outputText').innerText;

        navigator.clipboard.writeText(textToCopy).then(() => {

            alert('文案複製成功！資料也已經安穩地躺在儲值表裡了。');

        });

    }
